    LeetCode 3¶

    题目描述¶

    给定一个字符串 s ，请你找出其中不含有重复字符的 最长 子串的长度。

    示例 1:

    输入: s = "abcabcbb"

    输出: 3

    解释: 因为无重复字符的最长子串是 "abc"，所以其长度为 3。

    示例 2:

    输入: s = "bbbbb"

    输出: 1

    解释: 因为无重复字符的最长子串是 "b"，所以其长度为 1。

    示例 3:

    输入: s = "pwwkew"

    输出: 3

    解释: 因为无重复字符的最长子串是 "wke"，所以其长度为 3。

    请注意，你的答案必须是 子串 的长度，"pwke" 是一个子序列，不是子串。

    解题思路¶

    • 创建一个ASCII字符集大小的数组，用于存储字符最后出现的位置。
    • 使用两个指针，start 和 end
    • 刚开始的时候，start 和 end 都指向字符串的开始位置。
    • 使用 end 指针遍历字符串，同时将遇到的字符及其索引存入ascii数组中。
    • 如果ascii数组中已经存在当前 end 指针指向的字符，则需要移动 start 指针，直到该字符被移出，此时，得到了两个重复字符之间的数组大小
    • 在每一步，更新最长无重复字符子串的长度，该长度等于 end - start + 1。
    • 如果出现重复的情况，在while中更新start指针的位置，直到没有重复的字符。（此时相当于是重置了start和end指针的位置，使得长度重新归0）
    • 最后用max函数返回最长无重复字符子串的长度。

    int lengthOfLongestSubstring(char *s) {
    int maxLength = 0;
    int start = 0; // 窗口的起始位置
    int charIndex[128] = {0}; // 假设ASCII字符集，用于存储字符最后出现的位置

    for (int end = 0; s[end] != '\0'; end++) {
        // 如果字符已存在，移动start指针到重复字符的下一个位置
        while (charIndex[(int)s[end]] != 0) {
            charIndex[(int)s[start]] = 0; // 从哈希表中删除start指针的字符
            start++;
        }
        // 更新字符的索引
        charIndex[(int)s[end]] = end + 1; // 记录字符最后出现的位置
        // 更新最长子串的长度
        maxLength = max(maxLength, end - start + 1);
    }

    return maxLength;
}

int max(int a, int b) {
    return (a > b) ? a : b;
}

    可变数组¶

    数组是无法改变大小的，传统方法是搞一个大的数组，现在考虑说自己用C实现一个可以改变大小的数组

    1. 可以长大
    2. 有一个办法知道现在有多大（之前可以用sizeof(),但是现在不行了）
    3. 可以访问其中的单元

    typedef struct{
    int *array;
    int size;
}Array;//定义array结构体
Array array_create(int init_size){
    Array a;
    a.size = init_size;
    a.array = (int *)malloc(sizeof(int) * a.size);
    return a;
}
void array_free(Array *a){
    free(a->array);
    a->array = NULL;
    a->size = 0;
}
int array_size(const Array *a){
    return a->size;//调用size
}
int* array_at(Array *a, int index){
    return &a->array[index];//直接调用index位置的元素
}
void array_inflate(Array *a, int more_size){
    int *p = (int *)malloc(sizeof(int) * (a->size + more_size));
    //重新申请一个更大的数组，将原来的数组拷贝到新的数组中
    for(inti = 0; i < a->size; i++){
        p[i] = a->array[i];
    }
    //for循环拷贝数组
    free(a->array);//释放原来的数组
    a->array = p;//将新的数组赋给a->array,把新地址赋值给a->array这个指针

    a->size += more_size;//更新size
}

    ->¶

    在C语言中，a->array 表示通过指向结构体的指针 a 访问该结构体中的成员 array。

    具体解释 ：

    1. a 是指针：这里的 a 是一个指向某个结构体的指针。

    2. -> 是成员访问运算符：-> 运算符用于通过指针访问结构体或联合体中的成员。它等价于 (*a).array，只是 -> 更加简洁。

    3. array 是结构体的成员：array 是该结构体中的一个成员，可能是数组、指针、或者其他类型的变量。

    可变数组的缺陷¶

    每一次长大的时候都要申请一块新的内存空间，问题是拷贝要花时间，花内存。

    可能出现一种情况，明明有更大的内存但是没有足够的空间了

    可以采用linked block的方法，每次申请一个block，每个block有一个指针指向下一个block，这样就可以做到让一个数组指向下一个数组，再不够了再指向下一个数组，这样就可以实现一个无限长的数组。不断生长

    可以有这么一个东西，一部分是数据，另外一部分是指针，指针可以指向下一个同样的东西，下一个还是一个数据，一个指针（指向下一个）最后一个只一个结尾标志，没有下一个了

    还有一个头，带着数据和指针的是节点，这就是linkedlist 链表

    一般会这样定义

    这样定义了一个节点，里面有数据和指针，指针指向下一个节点，最后一个节点的指针指向NULL，表示结束。

    #include <stdio.h>

typedef struct Node {
    int value;
    struct Node *next;
} Node;

int main() {
    Node *head = NULL;
    int num;
    while (scanf("%d", &num) != EOF && num != -1) {
        if (num != -1) {
            //add to linked list,一开始整个linkedlist也是空的，需要add到linkedlist中
            Node *p = (Node *)malloc(sizeof(Node));//申请一个新的节点
            p->value = num;//赋值
            p->next = NULL;//最后一个节点的指针指向NULL
            //找到最后那个节点，然后把新的节点加到最后一个节点的后面
            Node *last = head;//从头开始找
            if (last == NULL) {//如果头是空的，那么新的节点就是头
                head = p;//第一个节点，直接让head指向新节点
            } else {
                while (last->next!= NULL) {//找到最后一个节点

                    last = last->next;//找到最后一个节点，最后一个节点的指针指向NULL
                }
                last->next = p;//最后一个节点的指针指向新的节点
            }
        }
    }
    for (Node *p = head; p != NULL; p = p->next) {
        printf("%d ", p->value);
    }
    return 0;
}

    这样就可以实现一个可变数组，但是链表的优势在于可以动态的增加元素，而且不需要申请新的内存空间，所以更加高效。

    void add(Node **head, int value) {
    //传入head的指针的指针
    Node *p = (Node *)malloc(sizeof(Node));
    p->value = value;
    p->next = NULL;
    Node *last = *head;
    if (last == NULL) {
        *head = p;
    } else {
        while (last->next != NULL) {
            last = last->next;
        }
        last->next = p;
    }
    return head;
}

    这样就可以在主函数中调用 add 函数，向链表中添加元素：

    void delete(Node **head, int value) {
    Node *p = *head;
    Node *prev = NULL;
    while (p!= NULL && p->value!= value) {
        prev = p;
        p = p->next;
    }
    if (p!= NULL) {
        if (prev == NULL) {
            *head = p->next;
        } else {
            prev->next = p->next;
        }
        free(p);
    }
}

    这个函数的作用是删除链表中值为 value 的节点，首先找到这个节点，然后把它的前一个节点的 next 指针指向它的下一个节点，然后释放这个节点的内存。

    这样就可以删除链表中的某个节点。

    void insert(Node **head, int value, int index) {
    Node *p = (Node *)malloc(sizeof(Node));
    p->value = value;
    p->next = NULL;
    if (index == 0) {
        p->next = *head;
        *head = p;
    } else {
        Node *prev = NULL;
        Node *curr = *head;
        int i = 0;
        while (curr!= NULL && i < index - 1) {
            prev = curr;
            curr = curr->next;
            i++;
        }
        if (curr == NULL) {
            printf("Index out of range\n");
        } else {
            p->next = curr;
            prev->next = p;
        }
    }
}

    这个函数的作用是插入一个值为 value 的节点到链表的第 index 个位置，首先申请一个新的节点，然后判断 index 是否为 0，如果是，那么把新的节点的 next 指针指向原来的头节点，然后把新的节点作为头节点，否则，需要找到第 index 个位置的节点，然后把新的节点插入到这个位置的前面。

    这样就可以在链表的任意位置插入一个节点。

    void merge(Node **head1, Node **head2) {
    Node *p1 = *head1;
    Node *p2 = *head2;
    Node *prev1 = NULL;
    Node *prev2 = NULL;
    while (p1!= NULL && p2!= NULL) {
        if (p1->value < p2->value) {
            prev1 = p1;
            p1 = p1->next;
        } else {
            prev2 = p2;
            p2 = p2->next;
        }
    }
    if (p1 == NULL) {
        prev2->next = p2;
    } else {
        prev1->next = p1;
    }
    *head1 = *head2;
}

    这个函数的作用是合并两个链表，首先找到两个链表的头节点，然后比较两个链表的第一个节点的值，把小的节点插入到大的节点的后面，直到两个链表的头节点有一个为空，然后把另一个链表的剩余节点插入到第一个链表的最后一个节点的后面，最后把第二个链表的头节点指向第一个链表的头节点，这样就合并了两个链表。

    这样就可以合并两个链表。

    void reverse(Node **head) {
    Node *p = *head;
    Node *prev = NULL;
    Node *next = NULL;
    while (p!= NULL) {
        next = p->next;
        p->next = prev;
        prev = p;
        p = next;
    }
    *head = prev;
}

    这个函数的作用是反转链表，首先找到链表的头节点，然后把它的 next 指针指向前一个节点，然后把当前节点指向前一个节点，直到链表的头节点为空，然后把链表的头节点指向最后一个节点，这样就反转了链表。

    这样就可以反转链表。

    /**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     struct ListNode *next;
 * };
 */
struct ListNode* addTwoNumbers(struct ListNode* l1, struct ListNode* l2) {
    struct ListNode *head = NULL, *tail = NULL;//声明head和tail两个指针，表示头和尾
    int carry = 0;//定义carry，表示进位
    while(l1||l2)//当两个存在null（任意一个到结尾的之前）
    {
        int n1 = l1 ? l1->val : 0;//如果存在li->val,就用li->val，否则就是0
        int n2 = l2 ? l2->val : 0;
        int sum = n1 + n2 +carry;//相加
        if(!head)//如果head是第一个（后面没有节点）
        {
            head = tail = malloc(sizeof(struct ListNode));//此时head = tail，我们可以直接在null后面申请一块结构体
            tail -> val = sum % 10;//然后赋值
            tail -> next = NULL;//下一个指向null
        }
        else
        {
            tail -> next = malloc(sizeof(struct ListNode));
            //本来指向null现在不指向null，指向新的结构体
            //tail->next指向下一个指向新的结构体
            tail -> next -> val = sum % 10;//给新的结构体赋值
            tail = tail -> next;//链接到tail上
            tail -> next = NULL;然后把新的tail变成null
        }
        carry = sum / 10;//统一在这里进位，避免重复
        if(l1)//如果l1存在，l1指向下一个节点
        {
            l1 = l1 -> next;
        }
        if(l2)//如果l2存在，l2指向下一个节点
        {
            l2 = l2 -> next;
        }

    }
    if(carry > 0)//如果最后还有进位，就再申请一个节点
    {
        tail -> next = malloc(sizeof(struct ListNode));
        tail -> next -> val = carry;
        tail -> next -> next = NULL;
    }

    return head;//返回头节点
}

    这个题解是用链表实现的，思路是用两个指针，一个指向l1，一个指向l2，然后相加，如果有进位，就把进位加到下一个节点上，直到两个链表都到结尾，然后把最后的进位加到最后一个节点上。

    bool isPalindrome(int x) {//判断是否为回文数
    if(x < 0){
        return false;//负数不会是回文数
    }
    long num = x;//存储x
    long num1 = 0;//用于储存x的反转数
    while(num != 0){
        int a = num % 10;//取num的个位
        num /= 10;//移除num的最后一个数字
        num1 = num1 * 10 + a;
    }
    if(num1 == x){
        return true;
    }else{
        return false;
    }
}
